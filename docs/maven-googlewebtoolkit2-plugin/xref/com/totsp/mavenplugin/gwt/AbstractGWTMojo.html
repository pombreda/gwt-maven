393    /**
394     * Helper hack for classpath problems, used as a fallback.
395     * 
396     * @return
397     */
398    protected ClassLoader fixThreadClasspath() {
399 
400       try {
401          ClassWorld world = new ClassWorld();
402 
403          //use the existing ContextClassLoader in a realm of the classloading space
404          ClassRealm root = world.newRealm("gwt-plugin", Thread.currentThread().getContextClassLoader());
405          ClassRealm realm = root.createChildRealm("gwt-project");
406 
407          for (Iterator it = BuildClasspathUtil.buildClasspathList(this, DependencyScope.COMPILE).iterator(); it
408                   .hasNext();) {
409             realm.addConstituent(((File) it.next()).toURI().toURL());
410          }
411 
412          Thread.currentThread().setContextClassLoader(realm.getClassLoader());
413          ///System.out.println("AbstractGwtMojo realm classloader = " + realm.getClassLoader().toString());
414 
415          return realm.getClassLoader();
416       }
417       catch (Exception e) {
418          e.printStackTrace();
419          throw new RuntimeException(e);
420       }
421    }

423    /**
424     * Create embedded GWT tomcat base dir based on properties.
425     * 
426     * @throws Exception
427     */
428    public void makeCatalinaBase() throws Exception {
429        getLog().debug("make catalina base for embedded Tomcat");        
430        
431        if (this.getWebXml() != null && this.getWebXml().exists()) {
432            this.getLog().info("source web.xml present - " + this.getWebXml() + " - using it with embedded Tomcat");
433        } else {
434            this.getLog().info("source web.xml NOT present, using default empty web.xml for shell");
435        }
436 
437        // note that MakeCatalinaBase (support jar) will use emptyWeb.xml if webXml does not exist 
438        String[] args = { this.getTomcat().getAbsolutePath(), this.getWebXml().getAbsolutePath(), this.getShellServletMappingURL() };
439        MakeCatalinaBase.main(args);
440 
441        if ((this.getContextXml() != null) && this.getContextXml().exists()) {
442            this.getLog().info("contextXml parameter present - " + this.getContextXml() + " - using it for embedded Tomcat ROOT.xml");
443            FileUtils.copyFile(this.getContextXml(), new File(this.getTomcat(), "conf/gwt/localhost/ROOT.xml"));
444        }
445    }
