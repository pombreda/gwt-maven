34  /**
35   * Runs the the project in the GWTShell for development.
36   * 
37   * Note that this goal is intended to be explicitly run from the command line 
38   * (execute phase=), whereas other GWT-Maven goals are not (others happen as 
39   * part of the standard Maven life-cycle phases: "compile" "test" "install").
40   * 
41   * @goal gwt
42   * @execute phase=compile
43   * @requiresDependencyResolution compile
44   * @description Runs the the project in the GWTShell for development.
45   * 
46   * @author ccollins
47   * @author cooper
48   */
49  public class GWTMojo extends AbstractGWTMojo {
50  
51      /** Creates a new instance of GWTMojo */
52      public GWTMojo() {
53          super();
54      }
55  
56      public void execute() throws MojoExecutionException, MojoFailureException {
57  
58          try {
59              this.makeCatalinaBase();
60          } catch (Exception e) {
61              throw new MojoExecutionException("Unable to build catalina.base", e);
62          }
63          if (!this.getOutput().exists()) {
64              this.getOutput().mkdirs();
65          }
66  
67          // build it for the correct platform
68          ScriptWriter writer = ScriptWriterFactory.getInstance();
69          File exec = writer.writeRunScript(this);        
70          
71          // run it
72          ScriptUtil.runScript(exec);
73      }
74  
75      /**
76       * Create embedded GWT tomcat base dir based on properties.
77       * 
78       * @throws Exception
79       */
80      public void makeCatalinaBase() throws Exception {
81          getLog().debug("make catalina base for embedded Tomcat");        
82          
83          if (this.getWebXml() != null && this.getWebXml().exists()) {
84              this.getLog().info("source web.xml present - " + this.getWebXml() + " - using it with embedded Tomcat");
85          } else {
86              this.getLog().info("source web.xml NOT present, using default empty web.xml for shell");
87          }
88  
89          // note that MakeCatalinaBase (support jar) will use emptyWeb.xml if webXml does not exist 
90          String[] args = { this.getTomcat().getAbsolutePath(), this.getWebXml().getAbsolutePath(), this.getShellServletMappingURL() };
91          MakeCatalinaBase.main(args);
92  
93          if ((this.getContextXml() != null) && this.getContextXml().exists()) {
94              this.getLog().info("contextXml parameter present - " + this.getContextXml() + " - using it for embedded Tomcat ROOT.xml");
95              FileUtils.copyFile(this.getContextXml(), new File(this.getTomcat(), "conf/gwt/localhost/ROOT.xml"));
96          }
97      }
98  }
