22  package com.totsp.mavenplugin.gwt;
23  
24  import java.io.File;
25  import java.util.Iterator;
26  
27  import org.apache.maven.model.Resource;
28  import org.apache.maven.plugin.MojoExecutionException;
29  import org.apache.maven.plugin.MojoFailureException;
30  
31  import com.totsp.mavenplugin.gwt.support.ExitException;
32  import com.totsp.mavenplugin.gwt.support.GwtWebInfProcessor;
33  import com.totsp.mavenplugin.gwt.util.FileIOUtils;
34  
35  /**
36   * Merges GWT servlet elements into deployment descriptor (and non GWT servlets into shell).
37   * 
38   * @goal mergewebxml
39   * @phase process-resources
40   * @requiresDependencyResolution compile
41   * @description Merges GWT servlet elements into deployment descriptor (and non GWT servlets into shell).
42   * 
43   * @author cooper
44   */
45  public class MergeWebXmlMojo extends AbstractGWTMojo {
46  
47      /** Creates a new instance of MergeWebXmlMojo */
48      public MergeWebXmlMojo() {
49          super();
50      }
51  
52      public void execute() throws MojoExecutionException, MojoFailureException {
53  
54          try {
55              this.getLog().info("copy source web.xml - " + this.getWebXml() + " to build dir (source web.xml required if mergewebxml execution is enabled)");
56              File destination = new File(this.getBuildDir(), "web.xml");
57              if (!destination.exists()) {
58                  destination.getParentFile().mkdirs();
59                  destination.createNewFile();
60              }
61              
62              FileIOUtils.copyFile(this.getWebXml(), destination);
63  
64              for (int i = 0; i < this.getCompileTarget().length; i++) {
65                  File moduleFile = null;
66                  for (Iterator it = this.getProject().getCompileSourceRoots().iterator(); it.hasNext()
67                          && moduleFile == null;) {
68                      File check = new File(it.next().toString() + "/" + this.getCompileTarget()[i].replace('.', '/')
69                              + ".gwt.xml");
70                      getLog().debug("Looking for file: " + check.getAbsolutePath());
71                      if (check.exists()) {
72                          moduleFile = check;
73                      }
74                  }
75                  for (Iterator it = this.getProject().getResources().iterator(); it.hasNext();) {
76                      Resource r = (Resource) it.next();
77                      File check = new File(r.getDirectory() + "/" + this.getCompileTarget()[i].replace('.', '/')
78                              + ".gwt.xml");
79                      getLog().debug("Looking for file: " + check.getAbsolutePath());
80                      if (check.exists()) {
81                          moduleFile = check;
82                      }
83                  }
84  
85                  this.fixThreadClasspath();
86  
87                  GwtWebInfProcessor processor = null;
88                  try {
89                      if (moduleFile != null) {
90                          getLog().info("Module file: " + moduleFile.getAbsolutePath());
91                          processor = new GwtWebInfProcessor(this.getCompileTarget()[i], moduleFile, destination
92                                  .getAbsolutePath(), destination.getAbsolutePath());
93                      } else {
94                          throw new MojoExecutionException("module file null");
95                      }
96                  } catch (ExitException e) {
97                      this.getLog().warn(e.getMessage());
98                      return;
99                  }
100                 processor.process();
101             }
102         } catch (Exception e) {
103             throw new MojoExecutionException("Unable to merge web.xml", e);
104         }
105     }
