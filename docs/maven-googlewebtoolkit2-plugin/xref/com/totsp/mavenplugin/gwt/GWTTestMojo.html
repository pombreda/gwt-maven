32  /*
33   * This is based on the clever work Will Pugh did in the original "gwtTest" stuff. 
34   * This has been refactored to make it a bit more robust, and to use the same write
35   * script approach all the other GWT-Maven mojos use (and those are used because
36   * they are easier to tweak and debug than the Maven classpath - unfortunately). 
37   * 
38   * Disclaimer: this is a giant hack because Surefire has some issues with GWTTestCase. 
39   *  
40   * Surefire states that it offers multiple ways to load the classpath
41   * (http://maven.apache.org/plugins/maven-surefire-plugin/examples/class-loading.html),
42   * but it doesn't seem to work for the plain java class path case. Manifest class path works, 
43   * and isolated classpath also works, but just getting to a plain java class path does not seem to work
44   * (surefire still refers to /tmp/surefireX). Without a plain java class path GWTTestCase
45   * won't work - because GWTTestCase inspects the classpath and sets itself up, and it 
46   * doesn't like anything other than a plain java classpath (doesn't like a manifest jar, or isolated). 
47   * Also, presuming surefire did work, in plain java class path mode, we would then still be 
48   * susceptible to the line too long on Windows issue surefire works around with the other modes.
49   * 
50   * A lot of research into just using surefire was done (we don't want to have to offer this special class for GWTTestCase) - after 
51   * RTFM and trying the useSystemClassLoader and useManifestOnlyJar settings, various ways, we still
52   * could never get it to work. These issues are similar to our experiences
53   *: http://www.mail-archive.com/users@maven.apache.org/msg87660.html - and 
54   * http://jira.codehaus.org/browse/SUREFIRE-508.
55   * 
56   * Hopefully we can kill this someday, it's a hack, but for now, this is the ONLY way 
57   * to run GWTTestCase based tests from an automated Maven build. 
58   */

60     public void execute() throws MojoExecutionException, MojoFailureException {
61        if (isSkip()) {
62           return;
63        }
64  
65        this.getLog().info("running GWTTestCase tests (using test name filter -  " + this.getTestFilter() + ")");
66  
67        FileWriter testResultsWriter = null;
68  
69        // build scripts for each test case for the correct platform
70        // (note that scripts end up in outputDirectory/gwtTest)
71        ScriptWriter writer = ScriptWriterFactory.getInstance();
72        writer.writeTestScripts(this);
73  
74        // run the scripts
75        boolean testFailure = false;
76        File testDir = new File(this.getBuildDir(), "gwtTest");
77        FileFilter fileFilter = new WildcardFileFilter("gwtTest-*");
78        File[] files = testDir.listFiles(fileFilter);
79        for (int i = 0; i < files.length; i++) {
80           File test = files[i];         
81  
82           // create results writer
83           try {
84              String outTestName = test.getName();
85              outTestName = outTestName.substring(0, test.getName().lastIndexOf(".")); // strip end .sh/cmd
86              outTestName = outTestName.substring(8, outTestName.length()); // strip start gwtTest-
87              testResultsWriter = new FileWriter(new File(testDir, "TEST-" + outTestName + ".txt"));
88  
89              // run test script and capture output
90              com.totsp.mavenplugin.gwt.scripting.TestResult testResult = ScriptUtil.runTestScript(test);
91  
92              // if testCode not success, overall must fail build
93              if (testResult.code == TestCode.ERROR || testResult.code == TestCode.FAILURE) {
94                 testFailure = true;
95              }
96  
97              // write results to result file
98              testResultsWriter.write("Test Code - " + testResult.code + "\n");
99              testResultsWriter.write("Test Output: \n" + testResult.message + "\n");
100             testResultsWriter.flush();
101             testResultsWriter.close();
102             
103             this.getLog().info(outTestName + " completed, GWTTestCase result: " + testResult.lastLine);
104          }
105          catch (IOException e) {
106             throw new MojoExecutionException("unable to create test results output file", e);
107          }
108       }
109 
110       // after the loop show output and or handle overall failure
111       // TODO add up results and show X runs - X successes, X failures, etc 
112 
113       this.getLog().info("all tests completed - ran " + files.length + " tests - see results in target/gwtTest");
114 
115       if (testFailure) {
116          throw new MojoExecutionException("There were GWTTestCase test failures - see results in target/gwtTest");
117       }
118    }
