23  package com.totsp.mavenplugin.gwt;
24  
25  import java.io.File;
26  
27  import org.apache.maven.plugin.MojoExecutionException;
28  import org.apache.maven.plugin.MojoFailureException;
29  
30  import com.totsp.mavenplugin.gwt.scripting.ScriptUtil;
31  import com.totsp.mavenplugin.gwt.scripting.ScriptWriter;
32  import com.totsp.mavenplugin.gwt.scripting.ScriptWriterFactory;
33  
34  /**
35   * Invokes the GWTCompiler for the project source.
36   * 
37   * @goal compile
38   * @phase process-classes
39   * @requiresDependencyResolution compile
40   * @description Invokes the GWTCompiler for the project source.
41   * 
42   * @author cooper
43   * @author ccollins
44   */
45  public class CompileMojo extends AbstractGWTMojo {
46  
47      /** Creates a new instance of CompileMojo */
48      public CompileMojo() {
49          super();
50      }
51  
52      public void execute() throws MojoExecutionException, MojoFailureException {
53          if (!this.getOutput().exists()) {
54              this.getOutput().mkdirs();
55          }
56  
57          // build it for the correct platform
58          ScriptWriter writer = ScriptWriterFactory.getInstance();
59          File exec = writer.writeCompileScript(this);        
60          
61          // run it
62          ScriptUtil.runScript(exec);
63      }
64  }
