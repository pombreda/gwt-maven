49      /**
50       * Write run script.
51       */
52      public File writeRunScript(AbstractGWTMojo mojo) throws MojoExecutionException {
53          String filename = (mojo instanceof DebugMojo) ? "debug.cmd" : "run.cmd";
54          File file = new File(mojo.getBuildDir(), filename);
55          PrintWriter writer = this.getPrintWriterWithClasspath(mojo, file, DependencyScope.RUNTIME);        
56          
57          String extra = (mojo.getExtraJvmArgs() != null) ? mojo.getExtraJvmArgs() : "";
58          writer.print("\"" + AbstractGWTMojo.JAVA_COMMAND + "\" " + extra + " -cp %CLASSPATH% ");
59  
60          if (mojo instanceof DebugMojo) {
61              writer.print(" -Xdebug -Xnoagent -Djava.compiler=NONE -Xrunjdwp:transport=dt_socket,server=y,address=");
62              writer.print(mojo.getDebugPort());
63              writer.print(",suspend=y ");
64          }
65  
66          writer.print("-Dcatalina.base=\"" + mojo.getTomcat().getAbsolutePath() + "\" ");
67          writer.print(" com.google.gwt.dev.GWTShell");
68          writer.print(" -gen \"");
69          writer.print(mojo.getGen().getAbsolutePath());
70          writer.print("\" -logLevel ");
71          writer.print(mojo.getLogLevel());
72          writer.print(" -style ");
73          writer.print(mojo.getStyle());
74          writer.print(" -out ");
75          writer.print("\"" + mojo.getOutput().getAbsolutePath() + "\"");
76          writer.print(" -port ");
77          writer.print(Integer.toString(mojo.getPort()));
78  
79          if (mojo.isNoServer()) {
80              writer.print(" -noserver ");
81          }
82  
83          writer.print(" " + mojo.getRunTarget());
84          writer.println();
85          writer.flush();
86          writer.close();
87  
88          return file;
89      }

91      /**
92       * Write compile script.
93       */
94      public File writeCompileScript(AbstractGWTMojo mojo) throws MojoExecutionException {
95          File file = new File(mojo.getBuildDir(), "compile.cmd");
96          PrintWriter writer = this.getPrintWriterWithClasspath(mojo, file, DependencyScope.COMPILE);
97  
98          for (String target : mojo.getCompileTarget()) {
99              String extra = (mojo.getExtraJvmArgs() != null) ? mojo.getExtraJvmArgs() : "";
100             writer.print("\"" + AbstractGWTMojo.JAVA_COMMAND + "\" " + extra + " -cp %CLASSPATH% ");
101             writer.print(" com.google.gwt.dev.GWTCompiler ");
102             writer.print(" -gen \"");
103             writer.print(mojo.getGen().getAbsolutePath());
104             writer.print("\" -logLevel ");
105             writer.print(mojo.getLogLevel());
106             writer.print(" -style ");
107             writer.print(mojo.getStyle());
108 
109             writer.print(" -out ");
110             writer.print(mojo.getOutput().getAbsolutePath());
111             writer.print(" ");
112 
113             if (mojo.isEnableAssertions()) {
114                 writer.print(" -ea ");
115             }
116 
117             writer.print(target);
118             writer.println();
119         }
120 
121         writer.flush();
122         writer.close();
123 
124         return file;
125     }

171     /**
172      * Write test scripts.
173      */
174     public void writeTestScripts(AbstractGWTMojo mojo) throws MojoExecutionException {
175         
176         // get extras
177         String extra = (mojo.getExtraJvmArgs() != null) ? mojo.getExtraJvmArgs() : "";        
178         String testExtra = mojo.getExtraTestArgs() != null ? mojo.getExtraTestArgs() : ""; 
179         
180         // make sure output dir is present
181         File outputDir = new File(mojo.getBuildDir(), "gwtTest");
182         outputDir.mkdirs();
183         outputDir.mkdir();
184 
185         // for each test compile source root, build a test script
186         List<String> testCompileRoots = mojo.getProject().getTestCompileSourceRoots();
187         for (String currRoot : testCompileRoots) {
188 
189             Collection<File> coll = FileUtils.listFiles(new File(currRoot),
190                     new WildcardFileFilter(mojo.getTestFilter()), HiddenFileFilter.VISIBLE);
191 
192             for (File currFile : coll) {
193                 
194                 String testName = currFile.toString();                
195                 mojo.getLog().debug(("gwtTest test match found (after filter applied) - " + testName));
196 
197                 // parse off the extension
198                 if (testName.lastIndexOf('.') > testName.lastIndexOf(File.separatorChar)) {
199                     testName = testName.substring(0, testName.lastIndexOf('.'));
200                 }
201                 if (testName.startsWith(currRoot)) {
202                     testName = testName.substring(currRoot.length());
203                 }
204                 if (testName.startsWith(File.separator)) {
205                     testName = testName.substring(1);
206                 }
207                 testName = StringUtils.replace(testName, File.separatorChar, '.');
208                 mojo.getLog().debug("testName after parsing - " + testName);
209 
210                 // start script inside gwtTest output dir, and name it with test class name
211                 File file = new File(mojo.getBuildDir() + File.separator + "gwtTest", "gwtTest-" + testName + ".cmd");
212                 PrintWriter writer = this.getPrintWriterWithClasspath(mojo, file, DependencyScope.TEST);
213 
214                 // build Java command
215                 writer.print("\"" + AbstractGWTMojo.JAVA_COMMAND + "\" ");
216                 if (extra.length() > 0) {
217                     writer.print(" " + extra + " ");
218                 }
219                 if (testExtra.length() > 0) {
220                     writer.print(" " + testExtra + " ");
221                 }
222                 writer.print(" -cp %CLASSPATH% ");
223                 writer.print("junit.textui.TestRunner ");
224                 writer.print(testName);
225 
226                 // write script out                
227                 writer.flush();
228                 writer.close();
229             }
230         }      
231     }

233     /**
234      * Util to get a PrintWriter with Windows preamble.
235      * 
236      * @param mojo
237      * @param file
238      * @return
239      * @throws MojoExecutionException
240      */
241     private PrintWriter getPrintWriterWithClasspath(final AbstractGWTMojo mojo, File file, final DependencyScope scope)
242             throws MojoExecutionException {
243 
244         PrintWriter writer = null;
245         try {
246             writer = new PrintWriter(new FileWriter(file));
247             writer.println("@echo off");
248         } catch (IOException e) {
249             throw new MojoExecutionException("Error creating script - " + file, e);
250         }
251 
252         try {
253             Collection<File> classpath = BuildClasspathUtil.buildClasspathList(mojo, scope);
254             writer.print("set CLASSPATH=");
255 
256             StringBuffer cpString = new StringBuffer();
257 
258             for (File f : classpath) {
259                 cpString.append("\"" + f.getAbsolutePath() + "\";");
260 
261                 // break the line at 4000 characters to try to avoid max size
262                 if (cpString.length() > 4000) {
263                     writer.println(cpString);
264                     cpString = new StringBuffer();
265                     writer.print("set CLASSPATH=%CLASSPATH%;");
266                 }
267             }
268             writer.println(cpString);
269             writer.println();
270         } catch (DependencyResolutionRequiredException e) {
271             throw new MojoExecutionException("Error creating script - " + file, e);
272         }
273 
274         writer.println();
275         return writer;
276     }
