36  public class BuildClasspathUtil {

38      /**
39       * Build classpath list using either gwtHome (if present) or using *project*
40       * dependencies. Note that this is ONLY used for the script/cmd writers (so
41       * the scopes are not for the compiler, or war plugins, etc).
42       * 
43       * This is required so that the script writers can get the dependencies they need
44       * regardless of the Maven scopes (still want to use the Maven scopes for everything else
45       * Maven, but for GWT-Maven we need to access deps differently - directly at times).
46       * 
47       * @author ccollins
48       * 
49       * @param mojo
50       * @param scope
51       * @return file collection for classpath
52       * @throws DependencyResolutionRequiredException
53       */
54      public static Collection<File> buildClasspathList(final AbstractGWTMojo mojo, final DependencyScope scope)
55              throws DependencyResolutionRequiredException, MojoExecutionException {
56  
57          mojo.getLog().info("establishing classpath list (buildClaspathList - scope = " + scope + ")");
58  
59          File gwtHome = mojo.getGwtHome();
60          MavenProject project = mojo.getProject();
61  
62          Set<File> items = new LinkedHashSet<File>();
63  
64          // inject GWT jars and relative native libs for all scopes
65          // (gwt-user and gwt-dev should be scoped provided to keep them out of
66          // other maven stuff - not end up in war, etc - this util is only used for GWT-Maven scripts)
67          // TODO filter the rest of the stuff so we don't double add these
68          if (gwtHome != null) {
69              mojo.getLog().info(
70                      "google.webtoolkit.home (gwtHome) set, using it for GWT dependencies - "
71                              + gwtHome.getAbsolutePath());
72              items.addAll(BuildClasspathUtil.injectGwtDepsFromGwtHome(gwtHome, mojo));
73          } else {
74              mojo.getLog().info("google.webtoolkit.home (gwtHome) *not* set, using project POM for GWT dependencies");
75              items.addAll(BuildClasspathUtil.injectGwtDepsFromRepo(mojo));
76          }
77  
78          // add sources and resources
79          if (mojo.getSourcesOnPath()) {
80              for (Iterator it = project.getCompileSourceRoots().iterator(); it.hasNext();) {
81                  items.add(new File(it.next().toString()));
82              }
83              for (Iterator it = project.getResources().iterator(); it.hasNext();) {
84                  Resource r = (Resource) it.next();
85                  items.add(new File(r.getDirectory()));
86              }
87          }
88  
89          // add classes dir
90          items.add(new File(project.getBasedir(), "classes"));
91  
92          // if runtime add runtime
93          if (scope == DependencyScope.RUNTIME) {
94              for (Iterator it = project.getRuntimeClasspathElements().iterator(); it.hasNext();) {
95                  items.add(new File(it.next().toString()));
96              }            
97          }
98          
99          // if test add test
100         if (scope == DependencyScope.TEST) {
101             for (Iterator it = project.getTestClasspathElements().iterator(); it.hasNext();) {
102                 items.add(new File(it.next().toString()));
103             }
104             
105             // add test sources and resources
106             for (Iterator it = project.getTestCompileSourceRoots().iterator(); it.hasNext();) {
107                 items.add(new File(it.next().toString()));
108             }
109             for (Iterator it = project.getTestResources().iterator(); it.hasNext();) {
110                 Resource r = (Resource) it.next();
111                 items.add(new File(r.getDirectory()));
112             }
113         } 
114         
115         // add compile (even when scope is other than)
116         for (Iterator it = project.getCompileClasspathElements().iterator(); it.hasNext();) {
117             items.add(new File(it.next().toString()));
118         }
119 
120         // add system 
121         for (Iterator it = project.getSystemClasspathElements().iterator(); it.hasNext();) {
122             items.add(new File(it.next().toString()));
123         }        
124 
125         mojo.getLog().debug("SCRIPT INJECTION CLASSPATH LIST");
126         for (File f : items) {
127             mojo.getLog().debug("   " + f.getAbsolutePath());
128         }
129 
130         return items;
131     }

133     /**
134      * Helper to inject gwt-user and gwt-dev into classpath from gwtHome, ONLY
135      * for compile and run scripts.
136      * 
137      * @param mojo
138      * @return
139      */
140     public static Collection<File> injectGwtDepsFromGwtHome(final File gwtHome, final AbstractGWTMojo mojo) {
141         mojo
142                 .getLog()
143                 .debug(
144                         "injecting gwt-user and gwt-dev for script classpath from google.webtoolkit.home (and expecting relative native libs)");
145         Collection<File> items = new LinkedHashSet<File>();
146         File userJar = new File(gwtHome, "gwt-user.jar");
147         File devJar = new File(gwtHome, ArtifactNameUtil.guessDevJarName());
148         items.add(userJar);
149         items.add(devJar);
150         return items;
151     }

153     /**
154      * Helper to inject gwt-user and gwt-dev into classpath from repo, ONLY for
155      * compile and run scripts.
156      * 
157      * @param mojo
158      * @return
159      */
160     public static Collection<File> injectGwtDepsFromRepo(final AbstractGWTMojo mojo) throws MojoExecutionException {
161         mojo
162                 .getLog()
163                 .debug(
164                         "injecting gwt-user and gwt-dev for script classpath from local repository (and expecting relative native libs)");
165         Collection<File> items = new LinkedHashSet<File>();
166 
167         Artifact gwtUser = mojo.getArtifactFactory().createArtifactWithClassifier("com.google.gwt", "gwt-user",
168                 mojo.getGwtVersion(), "jar", null);
169         Artifact gwtDev = mojo.getArtifactFactory().createArtifactWithClassifier("com.google.gwt", "gwt-dev",
170                 mojo.getGwtVersion(), "jar", ArtifactNameUtil.getPlatformName());
171 
172         try {
173             mojo.getResolver().resolve(gwtUser, null, mojo.getLocalRepository());
174             mojo.getResolver().resolve(gwtDev, null, mojo.getLocalRepository());
175             items.add(gwtUser.getFile());
176             items.add(gwtDev.getFile());
177         } catch (ArtifactNotFoundException e) {
178             throw new MojoExecutionException("artifact not found - " + e.getMessage(), e);
179         } catch (ArtifactResolutionException e) {
180             throw new MojoExecutionException("artifact resolver problem - " + e.getMessage(), e);
181         }
182 
183         return items;
184     }
